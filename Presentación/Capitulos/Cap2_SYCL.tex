\begin{frame}{SYCL}
% \begin{center}
SYCL is a \textbf{parallel focused} abstraction layer created for C++

Characteristics:
\begin{columns}
  \begin{column}{0.48\textwidth}
    \begin{itemize}
      \item High-level
      \item Standard and modern C++
      \item Single source
      \item Portable code \\~\\
    \end{itemize}
  \end{column}
  \begin{column}{0.48\textwidth}
    \begin{center}
      \includegraphics[width=0.7\linewidth]{Images/sycl-logo.png}
    \end{center}
  \end{column}
\end{columns}

  Targets a range of heterogeneous devices and backends
\end{center}
\endblock{}
